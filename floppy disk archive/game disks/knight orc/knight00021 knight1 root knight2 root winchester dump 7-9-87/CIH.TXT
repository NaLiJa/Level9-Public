.CHECKIFSCENERYHERE
; given OBJECT as a noun requested by user (relative to NOUNOFFSET)
; see if it is part of a room description, and
; if so, return RESULT=TRUE if in current room
 gosub @CalcTerrainAndTree
; x1=tree type, x2+x3=terrain type
 add x3,x2 ; x3:=terrain type
 x2=601 ; 900-nounoffset
 add x2,x1
 if x2=object then sceneryhere
 x2=604
 add x2,x3 ; add on terrain type(x3)
 if x2<>object then returnfalse
; drop through to return TRUE
.sceneryhere
 processed=true
 goto @returntrue
;---
